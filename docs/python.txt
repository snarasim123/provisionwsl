alpine python compile/install
-----------------------------

PYTHON_VERSION="3.8.9"
PYTHON_MAJOR_VERSION="3.8"

sudo apk add --no-cache openssh
sudo apk del --no-cache ansible aws-cli
sudo apk del --no-cache py3-passlib 

sudo apk add --no-cache build-base wget libffi-dev  libbz2 libffi zlib zlib-dev bzip2-dev readline-dev ncurses-dev xz-dev  ca-certificates 
#tk-dev  

cd /tmp
wget https://www.python.org/ftp/python/${PYTHON_VERSION}/Python-${PYTHON_VERSION}.tgz
tar -xvzf Python-${PYTHON_VERSION}.tgz
cd Python-${PYTHON_VERSION}
./configure --prefix=/usr/local
make
sudo make install

sudo ln -sf "/usr/local/bin/python${PYTHON_MAJOR_VERSION}" /usr/bin/python3
sudo ln -sf "/usr/local/bin/python${PYTHON_MAJOR_VERSION}" /usr/bin/python
sudo ln -sf /usr/local/bin/pip3 /usr/bin/pip3

sudo apk add py3-passlib ansible aws-cli
